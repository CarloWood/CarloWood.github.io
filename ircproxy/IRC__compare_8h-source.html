00026 //! @brief Compare two IRC characters in a 'case insensitive' way.
00027 //
00028 // Returns whether c1 is lexicographically less than (-1), equal (0) or larger than (1) c2.
00029 inline int IRCStringCompare(char c1, char c2)
00030 {
00031   int lc1 = static_cast<unsigned char>(ToLower(c1));
00032   int lc2 = static_cast<unsigned char>(ToLower(c2));
00033   if (lc1 < lc2)
00034     return -1;
00035   if (lc1 > lc2)
00036     return 1;
00037   return 0;
00038 }
