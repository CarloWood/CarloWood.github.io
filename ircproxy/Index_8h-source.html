00014 //! @brief Index generator for instances of type INDEX_POOL_TYPE.
00015 //
00016 // INDEX_POOL_TYPE must be derived from Index.
00017 // For each instance of class INDEX_POOL_TYPE index() will
00018 // return an integer unique for that instance that will
00019 // be constant throughout the life-time of that instance.
00020 template<typename INDEX_POOL_TYPE>
00021   class Index {
00022     public:
00023       static int const max_index = 8 * sizeof(long);            //!< The maximum number of allowed INDEX_POOL_TYPE instances.
00024 
00025     private:
00026       static long S_index_pool;                                 //!< A mask to keep track of used indices (a set bit marks a FREE index).
00027       int const M_index;                                        //!< The index of this instance.
00028 
00029     protected:
00030       //! Construct an Index base class, automatically setting M_index to a free index.
00031       Index(void) : M_index(ffsl(S_index_pool) - 1) { ASSERT(M_index != -1); S_index_pool &= ~(1UL << M_index); }
00032       //! Destructor. Frees up the index for reuse.
00033       ~Index(void) { S_index_pool |= (1 << M_index); }
00034 
00035     public:
00036       //! Return a small integer, unique for the instance and constant throughout it's life time.
00037       int index(void) const { return M_index; }
00038 
00039       //! Return the size of a vector that would be needed for the largest index currently in use.
00040       static int size(void) { int r = 0; unsigned long i = (~static_cast<unsigned long>(S_index_pool)) << 1; while(i >>= 1) ++r; return r; }
00041 
00042     private:
00043       static void sanity_check(void)
00044           {
00045             // INDEX_POOL_TYPE must be derived from Index.
00046             (void)static_cast<Index*>((INDEX_POOL_TYPE*)0);
00047           }
00048   };
