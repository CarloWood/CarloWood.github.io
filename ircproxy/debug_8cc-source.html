00065     /*!
00066      * @internal
00067      * @brief Map containing the default debug channel states used at the start of each new thread.
00068      *
00069      * The first thread calls main, which calls debug::init which will initialize this
00070      * map with all debug channel labels and whether or not they were turned on in the
00071      * rcfile or not.
00072      */
00073     rcfile_dc_states_type rcfile_dc_states;
00074 
00075     /*!
00076      * @internal
00077      * @brief Set the default state of debug channel \a dc_label.
00078      *
00079      * This function is called once for each debug channel.
00080      */
00081     void set_state(char const* dc_label, bool is_on)
00082     {
00083       std::pair<rcfile_dc_states_type::iterator, bool> res =
00084           rcfile_dc_states.insert(rcfile_dc_states_type::value_type(std::string(dc_label), is_on));
00085       if (!res.second)
00086         Dout(dc::warning, "Calling set_state() more than once for the same label!");
00087       return;
00088     }
00089 
00090     /*!
00091      * @internal
00092      * @brief Save debug channel states.
00093      *
00094      * One time initialization function of rcfile_dc_state.
00095      * This must be called from debug::init after reading the rcfile.
00096      */
00097     void save_dc_states(void)
00098     {
00099       // We may only call this function once: it reflects the states as stored
00100       // in the rcfile and that won't change.  Therefore it is not needed to
00101       // lock `rcfile_dc_states', it is only written to by the first thread
00102       // (once, via main -> init) when there are no other threads yet.
00103       static bool second_time = false;
00104       if (second_time)
00105       {
00106         Dout(dc::warning, "Calling save_dc_states() more than once!");
00107         return;
00108       }
00109       second_time = true;
00110       ForAllDebugChannels( set_state(debugChannel.get_label(), debugChannel.is_on()) );
00111     }

00115   /*!
00116    * @internal
00117    * @brief Returns the the original state of a debug channel.
00118    *
00119    * For a given \a dc_label, which must be the exact name (<tt>channel_ct::get_label</tt>) of an
00120    * existing debug channel, this function returns \c true when the corresponding debug channel was
00121    * <em>on</eM> at the startup of the application, directly after reading the libcwd runtime
00122    * configuration file (.libcwdrc).
00123    *
00124    * If the label/channel did not exist at the start of the application, it will return \c false
00125    * (note that libcwd disallows adding debug channels to modules - so this would probably
00126    * a bug).
00127    */
00128   bool is_on_in_rcfile(char const* dc_label)
00129   {
00130     rcfile_dc_states_type::const_iterator iter = rcfile_dc_states.find(std::string(dc_label));
00131     if (iter == rcfile_dc_states.end())
00132     {
00133       Dout(dc::warning, "is_on_in_rcfile(\"" << dc_label << "\"): \"" << dc_label << "\" is an unknown label!");
00134       return false;
00135     }
00136     return iter->second;
00137   }

00139   /*! @brief Initialize debugging code from new threads.
00140    *
00141    * This function needs to be called at the start of each new thread,
00142    * because a new thread starts in a completely reset state.
00143    *
00144    * The function turns on all debug channels that were turned on
00145    * after reading the rcfile at the start of the application.
00146    * Furthermore it initializes the debug ostream, its mutex and the
00147    * margin of the default debug object (Dout).
00148    */
00149   void init_thread(void)
00150   {
00151     // Turn on all debug channels that are turned on as per rcfile configuration.
00152     ForAllDebugChannels(
00153         if (!debugChannel.is_on() && is_on_in_rcfile(debugChannel.get_label()))
00154           debugChannel.on();
00155     );
00156 
00157     // Turn on debug output.
00158     Debug( libcw_do.on() );
00159 #if LIBCWD_THREAD_SAFE
00160     Debug( libcw_do.set_ostream(&std::cout, &cout_mutex) );
00161 #else
00162     Debug( libcw_do.set_ostream(&std::cout) );
00163 #endif
00164 
00165     static bool first_thread = true;
00166     if (!first_thread)                  // So far, the application has only one thread.  So don't add a thread id.
00167     {
00168       // Set the thread id in the margin.
00169       char margin[12];
00170       sprintf(margin, "%-10lu ", pthread_self());
00171       Debug( libcw_do.margin().assign(margin, 11) );
00172     }
00173   }

00231   /*! @brief Write a hexadecimal and ascii dump of a buffer to a given debug channel.
00232    *
00233    * This function can be used to print a binary block of data of \a size bytes, starting at \a buf.
00234    * The output is written to debug channel \a channel.
00235    */
00236   void dump_hex(libcwd::channel_ct const& channel, unsigned char const* buf, size_t size)
00237   {
00238     for (size_t addr = 0; addr < size; addr += 16)
00239     {
00240       LibcwDoutScopeBegin(DEBUGCHANNELS, libcwd::libcw_do, channel)
00241       LibcwDoutStream << std::hex << std::setfill('0') << std::setw(4) << addr << " |";
00242       int offset;
00243       for (offset = 0; offset < 16 && addr + offset < size; ++offset)
00244         LibcwDoutStream << ' ' << std::hex << std::setfill('0') << std::setw(2) << (int)buf[addr + offset];
00245       for (; offset < 16; ++offset)
00246         LibcwDoutStream << "   ";
00247       LibcwDoutStream << " | ";
00248       for (int offset = 0; offset < 16 && addr + offset < size; ++offset)
00249       {
00250         unsigned char c = buf[addr + offset];
00251         if (!std::isprint(c))
00252           c = '.';
00253         LibcwDoutStream << c;
00254       }
00255       LibcwDoutScopeEnd;
00256     }
00257   }
