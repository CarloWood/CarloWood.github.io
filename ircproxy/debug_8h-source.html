00116 /*! @brief Interface for marking scopes of invisible memory allocations.
00117  *
00118  * Creation of the object does nothing, you have to explicitly call
00119  * InvisibleAllocations::on.  Destruction of the object automatically
00120  * cancels any call to \c on of this object.  This makes it exception-
00121  * (stack unwinding) and recursive-safe.
00122  */
00123 struct InvisibleAllocations {
00124   int M_on;             //!< The number of times that InvisibleAllocations::on() was called.
00125   //! Constructor.
00126   InvisibleAllocations() : M_on(0) { }
00127   //! Destructor.
00128   ~InvisibleAllocations() { while (M_on > 0) off(); }
00129   //! Set invisible allocations on. Can be called recursively.
00130   void on(void) { libcwd::set_invisible_on(); ++M_on; }
00131   //! Cancel one call to on().
00132   void off(void) { assert(M_on > 0); --M_on; libcwd::set_invisible_off(); }
00133 };
