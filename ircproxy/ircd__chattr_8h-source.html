00030 //
00031 // Character attribute constants.
00032 //
00033 typedef unsigned int ntl_type;
00034 ntl_type const NTL_ALNUM  =  0x0001;  // (NTL_ALPHA | NTL_DIGIT)
00035 ntl_type const NTL_ALPHA  =  0x0002;  // (NTL_LOWER | NTL_UPPER)
00036 ntl_type const NTL_CNTRL  =  0x0004;  // \\000 - \\037 == 0x00 - 0x1F
00037 ntl_type const NTL_DIGIT  =  0x0008;  // 0123456789
00038 ntl_type const NTL_GRAPH  =  0x0010;  // (NTL_ALNUM | NTL_PUNCT)
00039 ntl_type const NTL_LOWER  =  0x0020;  // abcdefghijklmnopqrstuvwxyz{|}~
00040 ntl_type const NTL_PRINT  =  0x0040;  // (NTL_GRAPH | ' ')
00041 ntl_type const NTL_PUNCT  =  0x0080;  // !"#$%&'()*+,-./:;<=>?\@_`
00042 ntl_type const NTL_SPACE  =  0x0100;  // \\011\\012\\013\\014\\015\\040
00043 ntl_type const NTL_UPPER  =  0x0200;  // ABCDEFGHIJKLMNOPQRSTUVWXYZ[\\]^
00044 ntl_type const NTL_IRCCH  =  0x0400;  // Channel's names charset
00045 ntl_type const NTL_IRCCL  =  0x0800;  // Force toLower() in ch-name
00046 ntl_type const NTL_IRCNK  =  0x1000;  // Nick names charset, aka isvalid()
00047 ntl_type const NTL_IRCUI  =  0x2000;  // UserIDs charset, IRCHN plus tilde
00048 ntl_type const NTL_IRCHN  =  0x4000;  // Hostnames charset (weak, RFC 1033)
00049 ntl_type const NTL_IRCIP  =  0x8000;  // Numeric IPs charset (DIGIT and .)
00050 ntl_type const NTL_EOL    = 0x10000;  // \\r\\n
00051 ntl_type const NTL_KTIME  = 0x20000;  // Valid character for a k:line time
00052 ntl_type const NTL_CHPFX  = 0x40000;  // channel prefix char # & +
00053 ntl_type const NTL_IRCIP6 = 0x80000;  // Numeric IPv6 character (hex or colon)

00094 // Convert a character to its lower-case equivalent.
00095 inline char ToLower(char c)
00096 {
00097   return ToLowerTab_8859_1[c - std::numeric_limits<char>::min()];
00098 }
00099 
00100 // Convert a character to its upper-case equivalent.
00101 inline char ToUpper(char c)
00102 {
00103   return ToUpperTab_8859_1[c - std::numeric_limits<char>::min()];
