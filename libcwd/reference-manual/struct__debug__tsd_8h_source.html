56 struct debug_tsd_st {
57  friend class debug_ct;
58 
59 #if !LIBCWD_THREAD_SAFE
60  int _off;
61  // Debug output is turned on when this variable is -1, otherwise it is off.
62 #endif
63 
64  bool tsd_initialized;
65  // Set after initialization is completed.
66 
67 #if CWDEBUG_DEBUGOUTPUT
68  // Since we start with _off is -1 instead of 0 when CWDEBUG_DEBUG is set,
69  // we need to ignore the call to on() the first time it is called.
70  bool first_time;
71 #endif
72 
73  laf_ct* current;
74  // Current laf.
75 
76  std::ostream* current_bufferstream;
77  // The stringstream of the current laf.
78  // This should be set to current->bufferstream at all times.
79 
80  _private_::debug_stack_tst<laf_ct*> laf_stack;
81  // Store for nested debug calls.
82 
83  bool start_expected;
84  // Set to true when start() is expected, otherwise we expect a call to finish().
85 
86  bool unfinished_expected;
87  // Set to true when start() should cause a <unfinished>.
88 
89  int off_count;
90  // Number of nested and switched off continued channels till first switched on continued channel.
91 
92  _private_::debug_stack_tst<int> continued_stack;
93  // Stores the number of nested and switched off continued channels.
94 
95  debug_string_ct margin;
96  // The margin string.
97 
98  debug_string_ct marker;
99  // The marker string.
100 
101  debug_string_stack_element_ct* M_margin_stack;
102  // Pointer to list of pushed margins.
103 
104  debug_string_stack_element_ct* M_marker_stack;
105  // Pointer to list of pushed markers.
106 
107  unsigned short indent;
108  // Position at which debug message is printed.
109  // A value of 0 means directly behind the marker.
110 
111  // Accessed from LibcwdDout.
112  void start(debug_ct& debug_object, channel_set_data_st& channel_set LIBCWD_COMMA_TSD_PARAM);
113  void finish(debug_ct& debug_object, channel_set_data_st& /*channel_set*/ LIBCWD_COMMA_TSD_PARAM);
114  void fatal_finish(debug_ct& debug_object, channel_set_data_st& channel_set LIBCWD_COMMA_TSD_PARAM) __attribute__ ((__noreturn__));
115 
116  // Initialization and de-initialization.
117  void init();
118 #if LIBCWD_THREAD_SAFE
119  // In the non-threaded case, debug_ct contains a debug_tsd_st which
120  // may already be initialized before. Therefore don't initialize
121  // these in the non-threaded case, but rely on tsd_initialized,
122  // current_bufferstream and _off to be zeroed as a result of being
123  // part of a global object.
124  debug_tsd_st() : tsd_initialized(false), current_bufferstream(NULL) { }
125 #endif
126  ~debug_tsd_st();
127 };
