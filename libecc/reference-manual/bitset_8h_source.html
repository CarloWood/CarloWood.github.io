285 {

755  inline
756 #endif
757  void
758  bitset_base<N>::xor_with_zero_padded(bitset_base<m> const& bitset, int lsb, int msb, int shift)
759  {
760 #if defined(__i386__) && defined(HAVE_NASM)
761  libecc_shift_xorassign(this->vector, bitset.digits_ptr(), lsb, msb, shift);
762 #else
763  int digit1 = lsb >> bitset_digit_bits_log2;
764  int digit2 = msb >> bitset_digit_bits_log2;
765  bitset_digit_t d1 = 0;
766  // We can't use bitset.digit() because we need access to negative indexes.
767  bitset_digit_t const* const digits_ptr = bitset.digits_ptr();
768  if (shift < 0)
769  {
770  int bitshift = (-shift) & (bitset_digit_bits - 1);
771  int digitshift = (-shift) >> bitset_digit_bits_log2;
772  if (bitshift == 0)
773  {
774  for (int digit = digit2; digit >= digit1; --digit)
775  {
776  bitset_digit_t d2 = digits_ptr[digit];
777  this->vector[digit - digitshift] ^= d2;
778  }
779  }
780  else
781  {
782  for (int digit = digit2; digit >= digit1; --digit)
783  {
784  bitset_digit_t d2 = digits_ptr[digit];
785  this->vector[digit - digitshift] ^= (d2 >> bitshift) | (d1 << (bitset_digit_bits - bitshift));
786  d1 = d2;
787  }
788  this->vector[digit1 - 1 - digitshift] ^= (d1 << (bitset_digit_bits - bitshift));
789  }
790  }
791  else if (shift > 0)
792  {
793  int bitshift = shift & (bitset_digit_bits - 1);
794  int digitshift = shift >> bitset_digit_bits_log2;
795  if (bitshift == 0)
