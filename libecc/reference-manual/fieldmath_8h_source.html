85 template<class field_type, class square_functor>
86  field_type exponentiation(field_type const& base, mpz_class const& exponent,
87  square_functor const& do_square = multiplicative_square<field_type>())
88  {
89  field_type result(mpz_tstbit(exponent.get_mpz_t(), 0) ? base : field_type::unity());
90  field_type current_factor(base);
91  unsigned long next_bit = 1;
92  for(;;)
93  {
94  unsigned long current_bit = mpz_scan1(exponent.get_mpz_t(), next_bit);
95  if (current_bit == (unsigned long)-1)
96  break;
97  for(int cnt = current_bit - next_bit; cnt >= 0; --cnt)
98  do_square(current_factor);
99  result *= current_factor;
100  next_bit = current_bit + 1;
101  }
102  return result;
103  }

113 template<unsigned int m>
114  bitset<m>&
115  gcd(bitset<m>& polynomial_coefficients0, bitset<m>& polynomial_coefficients1)
116  {
117  bitset<m>* polycoef[2];
118  polycoef[0] = &polynomial_coefficients0;
119  polycoef[1] = &polynomial_coefficients1;
120  typename bitset<m>::const_reverse_iterator iter[2];
121 
122  // Determine the highest degree bit that is set, in the two polynomial_coefficients.
123  for (int p = 0; p < 2; ++p)
124  {
125  iter[p] = polycoef[p]->rbegin();
126  iter[p].find1();
127  if (iter[p] == polycoef[p]->rend())
128  return *polycoef[1 - p]; // The other polynomials is zero.
129  }
130  int distance = (iter[0] - iter[1]);
131  if (distance < 0)
132  distance = -distance;
133  int smallest = (iter[0] > iter[1]) ? 0 : 1;
134  for (int largest = 1 - smallest;; largest = 1 - largest, smallest = 1 - smallest)
135  {
136  bitset<m> temp(*polycoef[smallest]);
137  temp <<= distance;
138  int last_shift = distance;
139  *polycoef[largest] ^= temp;
140  for(int shift = distance - 1; shift >= 0; --shift)
141  {
142  if (!*(iter[largest] + (distance - shift)))
143  continue;
144  temp >>= (last_shift - shift);
145  last_shift = shift;
146  *polycoef[largest] ^= temp;
147  }
148  iter[largest].find1();
149  if (iter[largest] == polycoef[largest]->rend())
150  break; // Found the gcd.
151  distance = (iter[largest] - iter[smallest]);
152  }
153  return *polycoef[smallest];
154  }
