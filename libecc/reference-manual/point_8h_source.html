142 template<typename Polynomial, Polynomial const& a, Polynomial const& b>
143  void
144  point<Polynomial, a, b>::randomize(rds& random_source)
145  {
146  M_zero = false;
147  for(bool notdone = true; notdone;)
148  {
149  random_source.read(M_x.get_bitset());
150  bitset_digit_t x2buf[polynomial_type::square_digits];
151  polynomial_type& x2 = M_x.square(x2buf); // x^2
152  polynomial_type r((M_x + a) * x2 + b); // x^3 + a x^2 + b
153  try
154  {
155  M_y = polynomial_type(M_x, r);
156  notdone = false;
157  }
158  catch(std::domain_error const&)
159  {
160  }
161  }
162  bitset_digit_t digit;
163  random_source.read(&digit, 1);
164  if ((digit & 1))
165  {
166  if (!M_x.get_bitset().any())
167  M_zero = true;
168  else
169  M_y += M_x;
170  }
171  }

173 template<typename Polynomial, Polynomial const& a, Polynomial const& b>
174  bool
175  point<Polynomial, a, b>::check(void) const
176  {
177  if (M_zero)
178  return true;
179  bitset_digit_t x2buf[polynomial_type::square_digits];
180  polynomial_type& x2 = M_x.square(x2buf); // x^2
181  polynomial_type r((M_x + a) * x2 + b); // x^3 + a x^2 + b
182  bitset_digit_t y2buf[polynomial_type::square_digits];
183  polynomial_type& y2 = M_y.square(y2buf);
184  polynomial_type l(y2 + M_x * M_y); // y^2 + x y
185  return (r == l);
186  }

236 template<typename Polynomial, Polynomial const& a, Polynomial const& b>

238  point<Polynomial, a, b>::MULTIPLY_and_assign(point<Polynomial, a, b> const& pnt, mpz_class const& scalar)
239  {
240  point tmp(pnt);
241  M_zero = true;
242  unsigned long int p = 0;
243  size_t end = mpz_sizeinbase(scalar.get_mpz_t(), 2);
244  for (unsigned long int bit = mpz_scan1(scalar.get_mpz_t(), 0); bit < end; bit = mpz_scan1(scalar.get_mpz_t(), bit + 1))
245  {
246  for (; p < bit; ++p)
247  tmp += tmp;
248  *this += tmp;
249  }

252 template<typename Polynomial, Polynomial const& a, Polynomial const& b>
253  point<Polynomial, a, b>&
254  point<Polynomial, a, b>::operator+=(point<Polynomial, a, b> const& p1)
255  {
256  if (!p1.M_zero)
257  {
258  if (M_zero)
259  {
260  M_x = p1.M_x;
261  M_y = p1.M_y;
262  M_zero = false;
263  }
264  else if (M_x != p1.M_x)
265  {
266  polynomial_type s = M_y + p1.M_y;
267  s /= M_x + p1.M_x;
268  bitset_digit_t s2buf[polynomial_type::square_digits];
269  polynomial_type& new_x = s.square(s2buf);
270  new_x += s + M_x;
271  new_x += p1.M_x + a;
272  M_y += s * (M_x + new_x) + new_x;
273  M_x = new_x;
274  }
275  else if (M_y == p1.M_y && M_x.get_bitset().any())
276  {
277  polynomial_type s = M_x + M_y / M_x;
278  bitset_digit_t s2buf[polynomial_type::square_digits];
279  polynomial_type& new_x = s.square(s2buf);
280  new_x += s + a;
281  bitset_digit_t x2buf[polynomial_type::square_digits];
282  polynomial_type& x2 = M_x.square(x2buf);
283  M_y = x2 + new_x * (s + polynomial_type::unity());
284  M_x = new_x;
285  }
286  else
287  M_zero = true;
288  }
289  return *this;
290  }

292 template<typename Polynomial, Polynomial const& a, Polynomial const& b>
293  mpz_class
294  point<Polynomial, a, b>::order(order_algorithm algorithm) const
295  {
296  mpz_class n = 1;
297  switch(algorithm)
298  {
299  case order_brute_force:
300  {
301  point nP(*this);
302  while(!nP.M_zero && nP.M_x.get_bitset().any())
303  {
304  nP += *this;
305  ++n;
306  }
307  if (!nP.M_zero)
308  n += n;
309  break;
310  }
311  }
312  return n;
313  }
