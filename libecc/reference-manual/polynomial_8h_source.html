535  {

542  }
543  else
544  M_coefficients.flip(bit + k);
545  highbits.flip(bit);
546  }
547  }
548  }
549  }

551  {

553  {

555  {

559  M_coefficients.flip(bit + m - k);

656 template<unsigned int m, unsigned int k, unsigned int k1, unsigned int k2>
657  void
658  polynomial<m, k, k1, k2>::multiply_with(polynomial const& p1, bitset<m>& result) const
659  {
660  bitset_digit_t output[bitset<m>::digits * 2] __attribute__ ((aligned (8)));
661 
662  // Find the first non-zero digit in the input polynomial of this object.
663  unsigned int digit = 0;
664  while(M_coefficients.digit(digit) == 0) // Still zero?
665  {
666  output[digit] = 0; // That means that the output will end on zero too.
667  if (++digit == bitset<m>::digits)
668  {
669  result.reset(); // The whole polynomial is zero, the result will be zero too.
670  return;
671  }
672  }
673  unsigned int uninitialized_digit = digit; // The next digit of `output' that has not yet been initialized.
674  // Find the first digit in the input polynomial of this object whose first bit is set.
675  for(; digit < bitset<m>::digits; ++digit)
676  {
677  if ((M_coefficients.digit(digit) & 1)) // Is the first bit set?
678  {
679  // Set the output to p1 times this bit.
680  for (unsigned int d = 0; d < bitset<m>::digits; ++d)
681  output[d + digit] = p1.get_bitset().digit(d);
682  uninitialized_digit = bitset<m>::digits + digit;
683  ++digit; // Set to the next input digit.
684  break;
685  }
686  output[digit] = 0; // Initialize this digit of the output to 0.
687  ++uninitialized_digit;
688  }
689  // Set the remaining digits to zero, if any.
690  for(unsigned int remaining_digit = uninitialized_digit; remaining_digit < sizeof(output) / sizeof(bitset_digit_t); ++remaining_digit)
691  output[remaining_digit] = 0;
692  // Find for the remaining input digits the ones that have their first bit set.
693  for(; digit < bitset<m>::digits; ++digit)
694  if ((M_coefficients.digit(digit) & 1)) // Is the first bit set?
695  {
696  // Add p1 times this bit to the output.
697  for (unsigned int d = 0; d < bitset<m>::digits; ++d)
698  output[d + digit] ^= p1.get_bitset().digit(d);
699  }
700  // Create a bitset that will contain p1, shifted at most bitset_digit_bits - 1 to the left.
701  bitset<m + bitset_digit_bits - 1> shifted_p1;
702  // Start with having it shifted 1 bit to the left.
703  bitset_digit_t carry = 0;
704  unsigned int d = 0;
705  for(bitset_digit_t const* ptr = p1.get_bitset().digits_ptr(); ptr < p1.get_bitset().digits_ptr() + bitset<m>::digits; ++ptr, ++d)
706  {
707  shifted_p1.rawdigit(d) = (*ptr << 1) | carry;
708  carry = *ptr >> (8 * sizeof(bitset_digit_t) - 1);
709  }
710  if (d < bitset<m + bitset_digit_bits - 1>::digits)
711  shifted_p1.rawdigit(d) = carry;
712  for(bitset_digit_t bitmask = 2;;)
713  {
714  for(unsigned int digit = 0; digit < bitset<m>::digits; ++digit)
715  if ((M_coefficients.digit(digit) & bitmask))
716  {
717  for (unsigned int d = 0; d < shifted_p1.digits; ++d)
718  output[d + digit] ^= shifted_p1.digit(d);
719  }
720  bitmask <<= 1; // Next bit.
721  if (bitmask == 0) // Done?
722  break;
723  // Shift p1 one bit further to the left.
724  shifted_p1.template shift_op<1, left, assign>(shifted_p1);
725  }
726  // Reduce the resulting output of the multiplication.
727  reduce(output);
728  // Copy the reduced output to `result'.
729  std::memcpy(result.digits_ptr(), output, bitset<m>::digits * sizeof(bitset_digit_t));
730  }

777  // The following algorithm is based on the algorithm described on
778  // page 7 of http://research.sun.com/techrep/2001/smli_tr-2001-95.ps
779  // with significant optimization changes by Carlo Wood.
780  // The basis of the algorithm is to keep an invariants valid:
781  // A * y = U * x and B * y = V * x
782  // while reducing the size of A and B steadily to 0, until either is 1.
783  // The size of A and B is defined as the distance between the lowest
784  // and highest 1 in the bitset.
785 
786  // Make sure that there is enough space for a full bitset object
787  // and align the bitsets on a multiple of bitset_digit_t.
788  static unsigned int const digit_offset_UV = ((sizeof(bitset<m>) * 8 - 1) / bitset_digit_bits + 1);
789  static unsigned int const offset_UV = digit_offset_UV * bitset_digit_bits;
790  // Make room for exponents from at least t^-m till t^2m.
791  static unsigned int const digit_size_UV = 3 * digit_offset_UV;
792  // Variables A and B do not need this much space.
793  static unsigned int const digit_size_AB = bitset<m>::digits;
794  // One digit of padding, needed for xor_with_zero_padded.
795  static unsigned int const padding_digit_size = 1;
796  // 'F' (Floating-point polynomial) will be shifted to the right and
797  // is therefore defined to run from t^-2m till t^2m. This means it will
798  // be shifted OVER the other bitsets, but we don't need those anymore anyway.
799  static unsigned int const offset_F = 2 * offset_UV;
800  static unsigned int const size_F = 2 * m + offset_F;
801 
802  // Declare stack space for four variables.
803  // This array is going to be used as follows
804  // (where padding = padding_digit_size; AB = digit_size_AB and do_UV = digit_offset_UV):
805  //
806  // A:-> B:-> U:----> V:---->
807  // [padding][AB][padding][AB][padding][do_UV][do_UV][do_UV][padding][do_UV][do_UV][do_UV][padding]
808  // <-- digit_size_UV --> <-- digit_size_UV -->
809  // [do_UV] [do_UV]
810  // F:------------------------> F:------------------------>
811  // R:----> *or* R:---->
812  // <- offset_F -> <- offset_F ->
813  //
814  // This positioning is done to make the total number of bits that we have to clear in
815  // as small as possible. R is just an alias for either U or V depending on how the
816  // algorithm finishes. Note F overlaps with R and they alias eachother, F might also
817  // overlap with B but that variable isn't used anymore then.
818  //
819  // In order to avoid aliasing problems (see http://dbp-consulting.com/StrictAliasing.pdf),
820  // we put the bitsets in a struct together with bitset_digit_t for the padding and
821  // use a union to give F a place in all of this. Because objects with constructors
822  // aren't allowed in union's we have to use bitset_base.
823  structUV_bitpool_t {

1124 // Solve x^2 + b x = c.
1125 // Assuming that b != 0, there are 2 solutions: x1 and x1 + b.
1126 // This means that during the 'wiping' of the matrix in order
1127 // to solve x, one bit of x will stay undetermined. We need
1128 // to take special care to make sure that this will be a bit
1129 // for which a bit of 'b' is set, otherwise we'd return a wrong
1130 // value.
1131 //
1132 // If b equals zero, then the solution is the sqrt(c). Otherwise
1133 // we can devide both sides of the equation by b^2 and solve
1134 // y^2 + y = c/b^2, and set x = b * y.
1135 //
1136 // There will only be a solution to this equation iff 0 = Tr(c/b^2).
1137 // (simply square the equation m-1 times and add them all up).
1138 //
1139 // Note that if y1 is a solution, then so is y1 + 1, hence we
1140 // cannot determine the least significant bit of y.
1141 //
1142 // It is possible to compose a matrix A such that Ax = x^2 + x
1143 // because squaring is an automorphism of the field:
1144 // x is a sum of basis elements, ie x = b1 + b2 + b3 and
1145 // x^2 = b1^2 + b2^2 + b3^2. Therefore, if there exists a
1146 // matrix S such that Sb_i = b_i^2 for any basis element then
1147 // A = (S + I). Moreover, such a matrix S must exist because
1148 // there are exactly m basis elements, and a matrix of mxm
1149 // will always be able to satisfy that.
