233 AIPackagedTask<R(Args...)>::~AIPackagedTask()
234 {
235  // It should be impossible to destruct an AIPackagedTask while it is still
236  // executing when it is a member of parent_task; and that is the only way
237  // that this class should be used. The reason that is impossible is because the
238  // parent_task should be in a waiting state until we call m_task->signal(m_condition)
239  // in invoke() below, which we only do after m_phase is set to finished. Hence,
240  // the parent_task will not be destructed and therefore we won't be destructed either.
241  ASSERT(m_phase != executing);

281 // Called by parent task to dispatch the job to its own thread.
282 // After finishing the job, the parent will be signalled with
283 // m_condition set during construction.
284 //
285 // Returns true upon a successful queue; false when the queue is full
286 // in which case dispatch() can be tried again (a bit later).
287 template<typename R, typename ...Args>
288 bool AIPackagedTask<R(Args...)>::dispatch()
289 {
290  {
291  // Stop a new queue from being created while we're working with a queue, because that could move the queue.
292  AIThreadPool& thread_pool = AIThreadPool::instance();
293  auto queues_r = thread_pool.queues_read_access();
294  // Lock the queue.
295  auto& queue_ref = thread_pool.get_queue(queues_r, m_queue_handle);
296  {
297  auto queue = queue_ref.producer_access();
298  if (queue.length() == queue_ref.capacity())
299  {
300  m_phase = deferred;
301  return false;
302  }
303  // Pass job to thread pool.
304  m_phase = executing;
305  queue.move_in(std::function<bool()>([this](){ this->invoke(); return false; }));
306  } // Unlock queue.
307  // Now that we added something to queue, wake up one thread if needed.
308  queue_ref.notify_one();
309  } // And we're done with the queue, so also unlock AIThreadPool::m_queues.
310 
311  // Halt task until job finished.
312  wait_until([this](){ return m_phase == finished; }, m_condition);
313  return true;
314 }
