112  Node* prev = m_head.exchange(new_node, std::memory_order_acq_rel);

117  // | m_task -> task |

120  // If the previous value of m_head (prev) equal nullptr then we were
121  // the first and own the lock.
122  if (prev == nullptr)

124  Dout(dc::notice, "Mutex acquired, setting m_owner to " << new_node << " [" << task << "]");

130  // If prev is non-null then another task owned the lock at the moment
131  // we executed the above exchange. This task, when executing unlock(),
132  // will fail the compare_exchange_strong because we just changed the
