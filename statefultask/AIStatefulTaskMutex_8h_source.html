34 //
35 // A task mutex.
36 //
37 // Prevent different tasks from concurrently entering the same critical area.
38 //
39 // Consider an object that is shared between tasks, but may
40 // not be simultaneously accessed by two different threads.
41 //
42 // For example,
43 //
44 // // Required memory management.
45 // utils::MemoryPagePool mpp(0x8000);
46 // AIStatefulTaskMutex::init(&mpp);
47 // // A task mutex.
48 // AIStatefulTaskMutex m;
49 //
50 // Then multiple running tasks could use this to prevent concurrent access:
51 //
52 // ...
53 // case MyTask_lock:
54 // set_state(MyTask_locked);
55 // if (!m.lock(this, 1))
56 // {
57 // wait(1);
58 // break;
59 // }
60 // [[fallthrough]];
61 // case MyTask_locked:
62 // do_work();
63 // m.unlock();
64 //
65 class AIStatefulTaskMutex

95  // Try to obtain ownership for owner (recursive locking allowed).
96  // Returns true upon success and false upon failure to obtain ownership.
97  bool lock(AIStatefulTask* task, condition_type condition)
98  {
99  DoutEntering(dc::notice, "AIStatefulTaskMutex::lock(" << task << ", " << condition << ") [" << this << "]");
100 
101  Node* new_node = new (s_node_memory_resource.allocate(sizeof(Node))) Node;
102  std::atomic_init(&new_node->m_next, static_cast<Node*>(nullptr));
103  new_node->m_task = task;
104 
105 // Dout(dc::notice, "Create new node at " << new_node << " with m_next = " << new_node->m_next << "; m_task = " << new_node->m_task << " [" << task << "]");
106 
107  Node* prev = m_head.exchange(new_node, std::memory_order_acq_rel);
108 // Dout(dc::notice, "Replaced m_head with " << new_node << ", previous value of m_head was " << prev << " [" << task << "]");
109  // new_node:
110  // m_head -->.----------------.
111  // | m_next --------+--> nullptr
112  // | m_task -> task |
113  // `----------------'
114  //
115  // If the previous value of m_head (prev) equal nullptr then we were
116  // the first and own the lock.
117  if (prev == nullptr)
118  {
119  Dout(dc::notice, "Mutex acquired, setting m_owner to " << new_node << " [" << task << "]");
120  m_owner.store(new_node, std::memory_order_release);
121  return true;
122  }
123  Dout(dc::notice, "Mutex already locked by [" << task << "]");
124 
125  // If prev is non-null then another task owned the lock at the moment
126  // we executed the above exchange. This task, when executing unlock(),
127  // will fail the compare_exchange_strong because we just changed the
128  // value of m_head.
129  //
130  // Moreover that task then will spin lock on reading m_owner->m_next
131  // because that was initialized with nullptr when that task entered
132  // lock() and never changed.
133  //
134  // Below we set that m_next pointer to point to our node, so the
135  // task in unlock() can escape the spin lock.
136 // Dout(dc::notice, "Setting m_condition of new_node (" << new_node << ") to " << condition << " [" << task << "]");
137  new_node->m_condition = condition;
138 // Dout(dc::notice, "Setting m_next of prev (" << prev << ") to " << new_node << " [" << task << "]");
139  prev->m_next.store(new_node, std::memory_order_release);
140 
141  // Obtaining the lock failed. Halt the task
142  return false; // The caller must call task->wait(condition).
143  }

146  void unlock()
147  {
148  DoutEntering(dc::notice, "AIStatefulTaskMutex::unlock() [" << this << "]");
149 
150  Node* const owner = m_owner.load(std::memory_order_relaxed);
151 #ifdef CWDEBUG

153  AIStatefulTask* const task = owner->m_task;
154 #endif
155 // Dout(dc::notice, "Setting m_owner to nullptr, was " << owner << " [" << task << "]");
156  m_owner.store(nullptr, std::memory_order_relaxed);
157 // Dout(dc::notice, "Calling m_head.compare_exchange_strong(owner = " << owner << ", nullptr)");
158  Node* expected = owner;
159  if (m_head.compare_exchange_strong(expected, nullptr, std::memory_order_release, std::memory_order_relaxed))

161  Dout(dc::notice, "Success - m_head was reset to nullptr [" << task << "]");
162 // Dout(dc::notice, "deallocating " << owner << " [" << task << "]");
163  s_node_memory_resource.deallocate(owner);
164 // Dout(dc::notice, "Leaving unlock()");
165  return;
166  }

169  }
