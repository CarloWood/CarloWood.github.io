189  void notify_one() const
190  {
191  // Suppose that m_previous_total_reserved_threads is 2 here. That means that there are two threads
192  // that we're not allowed to use for this queue. Suppose that s_idle_threads equals 3. That means
193  // that there are three threads that can be woken up, but we're only allowed to wake up one.
194  // Since 3 > 2, the while condition is true and this thread enters the while loop.
195  // If at that point some other thread also enters this while loop and decrements s_idle_threads
196  // then the compare_exchange_weak will fail in this thread and we'd start from the start, this
197  // time with s_idle_threads equal to 2. If s_idle_threads is both decremented and incremented
198  // by yet another thread, then that's perfectly fine: in that case we are still allowed to wake
199  // up a thread for this queue.
200  int idle;
201  while ((idle = s_idle_threads.load(std::memory_order_relaxed)) > m_previous_total_reserved_threads) // This line takes 0.9...0.97 ns.
202  {
203  // Decrement s_idle_threads by one.
204  // Note that s_idle_threads is the only variable that is accessed outside the critical region of s_idle_mutex
205  // and there does not need to be synchronized with any other variable. Therefore it is enough to use
206  // std::memory_order_relaxed for all accesses to it.
207  if (!s_idle_threads.compare_exchange_weak(idle, idle - 1, std::memory_order_relaxed, std::memory_order_relaxed))
208  continue; // s_idle_threads changed in the meantime; try again from the start.
209  // A thread that reached this point is allowed to wake up one thread.
210  // A possible scenario could be:
211  // m_previous_total_reserved_threads
212  // Queue1: 1 reserved thread. 0
213  // Queue2: 1 reserved thread. 1
214  // Queue3: this thread/queue. 2
215  // Let s_idle_threads be 3.
216  // Thread A enters the while loop of queue3 (because 3 > 2)
217  // Thread B enters the while loop of queue1 (because 3 > 0)
218  // Thread C enters the while loop of queue1 (because 3 > 0)
219  // Thread D enters the while loop of queue2 (because 3 > 1)
220  // Thread A decrements s_idle_threads from 3 to 2.
221  // Thread B tries to decrement s_idle_threads, s_idle_threads isn't 3 anymore, so it starts at the top and enters the while loop of queue3 again (because 2 > 0).
222  // Thread C tries to decrement s_idle_threads, s_idle_threads isn't 3 anymore, so it starts at the top and enters the while loop of queue3 again (because 2 > 0).
223  // Thread D tries to decrement s_idle_threads, s_idle_threads isn't 3 anymore, so it starts at the top and enters the while loop of queue3 again (because 2 > 1).
224  // Thread B decrements s_idle_threads from 2 to 1.
225  // Thread C tries to decrement s_idle_threads, s_idle_threads isn't 2 anymore, so it starts at the top and enters the while loop of queue3 again (because 1 > 0).
226  // Thread D tries to decrement s_idle_threads, s_idle_threads isn't 2 anymore, so it starts at the top and fails to enter the while loop.
227  // Thread C decrements s_idle_threads from 1 to 0.
228  // Thread A obtains the lock on s_idle_mutex (this prevents that we call s_idle_cv.notify_one() while another
229  // thread is inbetween the increment of s_idle_threads and the call to s_idle_cv.wait()), calls notify_one() and wakes up a thread,
230  // and finally unlocks s_idle_mutex.
231  // Thread B and C to the same. So in total three threads are woken up.
232  // The first thread that is woken up will handle a task from Queue1, the second will handle a task from Queue1
233  // and the third will handle a task from Queue2. The task in Queue3 is not handled until all three woken
234  // threads are idle again (the first two being reserved for Queue1 and Queue2).
235  std::unique_lock<std::mutex> lk(s_idle_mutex);
236  ++s_to_be_woken;
237  s_idle_cv.notify_one(); // This lines turns out to take 19.5 microseconds!
238  break;
239  }
240  }

245  // A Worker is only const when we access it from a const worker_container_t.
246  // However, the (read) lock on the worker_container_t only protects the internals
247  // of the container, not its elements. So, all elements are considered mutable.
248  mutable std::thread m_thread;
249  mutable std::atomic_bool m_quit;
250 
251  // Construct a new Worker; do not associate it with a running thread yet.
252  // A write lock on m_workers is required before calling this constructor;
253  // that then blocks the thread from accessing m_quit until that lock is released
254  // so that we have time to move the Worker in place (using emplace_back()).
255  Worker(worker_function_t worker_function, int self) : m_thread(std::bind(worker_function, self)), m_quit(false) { }
256 
257  // The move constructor can only be called as a result of a reallocation, as a result
258  // of a size increase of the std::vector<Worker> (because Worker`s are put into it with
259  // emplace_back(), Worker is not copyable, and we never move a Worker out of the vector).
260  // That means that at the moment the move constuctor is called we have the exclusive
261  // write lock on the vector and therefore no other thread can access this Worker.
262  // Therefore it is safe to non-atomically copy m_quit (note that it cannot be moved or
263  // copied atomically).
264  Worker(Worker&& rvalue) : m_thread(std::move(rvalue.m_thread)), m_quit(rvalue.m_quit.load()) { rvalue.m_quit.store(true, std::memory_order_relaxed); }
