29 // DefaultMemoryPagePool is a globally accessible MemoryPagePool singleton,
30 // but with a life time equal of that of main().
31 //
32 // It must be initialized at the start of main before using of anything else
33 // from statefultask.
34 //
35 // For example,
36 //
37 // int main()
38 // {
39 // Debug(NAMESPACE_DEBUG::init();
40 //
41 // AIMemoryPagePool mpp; // Add this line at the top of main.
42 //
43 // ...
44 // }
45 //
46 // The above line is equivalent to one of
47 //
48 // statefultask::DefaultMemoryPagePool mpp{}; // The curly braces are required...
49 // statefultask::DefaultMemoryPagePool mpp(0x8000); // ...unless you pass at least one argument.
50 // statefultask::DefaultMemoryPagePool<utils::MemoryPagePool> mpp; // Or specify the (default) template parameter explicitly.
51 //
52 // In all cases AIMemoryPagePool::instance() becomes usable,
53 // which is currently only required when using AIStatefulTaskMutex (hence anything
54 // from filelock-task).
55 //
56 // Fine tuning is possible by passing the constructor arguments of a MemoryPagePool
57 // to DefaultMemoryPagePool, or to derive your own class from MemoryPagePool and pass
58 // that to DefaultMemoryPagePool as template argument. For example,
59 //
60 // statefultask::DefaultMemoryPagePool<MyMPP> default_memory_page_pool(constructor arguments of MyMPP);
61 //
62 // where MyMPP must be derived from utils::MemoryPagePool.
63 //
64 class DefaultMemoryPagePoolBase
65 {
66  private:
67  static utils::MemoryPagePool* s_instance;
68 
69  protected:
70  static void init(utils::MemoryPagePool* mpp)

72  // Only create a one DefaultMemoryPagePool object (at the top of main()).
73  ASSERT(s_instance == nullptr);

75  AIStatefulTaskMutex::init(mpp);
76  }
