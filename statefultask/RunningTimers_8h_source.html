37 // This is a tournament tree of queues of timers with the same interval.
38 //
39 // The advantage of using queues of timers with the same interval
40 // is that it is extremely cheap to insert a new timer with a
41 // given interval when there is already another timer with the
42 // same interval running. On top of that, it is likely that even
43 // a program that has a million timers running simultaneously
44 // only uses a handful of distinct intervals, so the size of
45 // the tree/heap shrinks dramatically.
46 //
47 // Assume the number of different intervals is 6, and thus tree_size == 8,
48 // then the structure of the data in RunningTimers could look like:
49 //
50 // Tournament tree (tree index:tree index)
51 // m_tree: 1:4
52 // / \ .
53 // 2:0 3:4
54 // / \ / \ .
55 // 4:0 5:3 6:4 7:6
56 // / \ / \ / \ / \ .
57 // m_cache: 18 no_timer 102 55 10 60 no_timer no_timer
58 // index of m_cache: 0 1 2 3 4 5 6 7
59 //
60 // Thus, m_tree[1] contains 4, m_tree[2] contains 0, m_tree[3] contains 4 and so on.
61 // Each time a parent contains the same interval (in) as one of its children and well
62 // such that m_cache[in] contains the smallest time_point value of the two.
63 // m_cache[in] contains a copy of the top of m_queues[in].
64 //

119  void increase_cache(int interval, Timer::time_point tp) // m_mutex must be locked.
120  {
121  ASSERT(tp >= m_cache[interval]);
122  m_cache[interval] = tp;
123 
124  int parent_ti = interval_to_parent_index(interval); // Let 'parent_ti' be the index of the parent node in the tree above 'interval'.
125 
126  int in = interval; // Let 'in' be the interval whose value is changed with respect to m_tree[parent_ti].
127  int si = in ^ 1; // Let 'si' be the interval of the current sibling of in.
128  for(;;)
129  {
130  Timer::time_point sv = m_cache[si];
131  if (tp > sv)
132  {
133  if (m_tree[parent_ti] == si)
134  break;
135  tp = sv;
136  in = si;
137  }
138  m_tree[parent_ti] = in; // Update the tree.
139  if (parent_ti == 1) // If this was the top-most node in the tree then we're done.
140  break;
141  si = m_tree[sibling_of(parent_ti)]; // Update the sibling interval.
142  parent_ti = parent_of(parent_ti); // Set 'parent_ti' to be the index of the parent node in the tree above 'parent_ti'.
143  }
144  }
145 
146  void decrease_cache(int interval, Timer::time_point tp) // m_mutex must be locked.
147  {
148  ASSERT(tp <= m_cache[interval]);
149  m_cache[interval] = tp; // Replace no_timer with tp.
150  // We just put a SMALLER value in the cache at position interval than what there was before.
151  // Therefore all we have to do is overwrite parents with our interval until the time_point
152  // value of the parent is less than tp.
153  int parent_ti = interval_to_parent_index(interval); // Let 'parent_ti' be the index of the parent node in the tree above 'interval'.
154  while (tp <= m_cache[m_tree[parent_ti]]) // m_tree[parent_ti] is the content of that node. m_cache[m_tree[parent_ti]] is the value.
155  {
156  m_tree[parent_ti] = interval; // Update that tree node.
157  if (parent_ti == 1) // If this was the top-most node in the tree then we're done.
158  break;
159  parent_ti = parent_of(parent_ti); // Set 'i' to be the index of the parent node in the tree above 'i'.
160  }
161  }
