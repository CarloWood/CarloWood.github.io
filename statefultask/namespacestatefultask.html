Convenience function to create tasks.

class ATask : public AIStatefulTask {
...
public:

};

{
boost::intrusive_ptr<ATask> m_task;
public:
SomeClass() : m_task(statefultask::create<ATask>(/* constructor arguments of ATask */)) { }
void initial_run(...)
{
m_task->init(...);
m_task->run(...);
}
};
