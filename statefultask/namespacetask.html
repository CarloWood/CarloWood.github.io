Convenience function to create tasks.

Typical usage,

class ATask : public AIStatefulTask {
...
public:
void init(...);
};
class SomeClass
{
boost::intrusive_ptr<ATask> m_task;
public:
SomeClass() : m_task(task::create<ATask>(/* constructor arguments of ATask */)) { }
void initial_run(...)
{
m_task->init(...);
m_task->run(...);
}
};
